function shareKakao() { const params = new URLSearchParams(window.location.search); // 1. URL에서 데이터 가져오기 ('ea' 낱개 파라미터 추가) const type = params.get('type') || ''; const method = params.get('method') || ''; const plt = params.get('plt') || '0'; const box = params.get('box') || '0'; const ea = params.get('ea') || '0'; // 낱개 데이터 const date = params.get('date') || ''; const worker = params.get('worker') || ''; // 사진 URL (일단 기본 이미지 유지) const imgUrl = params.get('img') || 'https://cdn-icons-png.flaticon.com/512/2765/2765089.png'; Kakao.Share.sendDefault({ objectType: 'feed', content: { // 제목 부분 title: `* [${type}] 작업 완료`, // 내용이 잘리지 않도록 작업일자를 설명 칸에 배치 description: `작업일자: ${date}`, imageUrl: imgUrl, link: { mobileWebUrl: 'https://www.appsheet.com', webUrl: 'https://www.appsheet.com', }, }, itemContent: { // 맨 위 상단 프로필 텍스트 (요청 1번 반영) profileText: '[입&출고 알림]', // 리스트 항목 (카카오 제한인 딱 5개로 맞춰서 절대 안 잘림!) items: [ { item: 'PLT', itemOp: plt }, { item: 'BOX', itemOp: box }, { item: '낱개', itemOp: ea }, { item: '작업방식', itemOp: method }, { item: '담당자', itemOp: worker } ] }, buttons: [ { title: '모바일에서 확인해주세요.', link: { mobileWebUrl: 'https://www.appsheet.com', webUrl: 'https://www.appsheet.com', }, }, ], }); }
